git clone <url>                 -- retrieve repo
git remote -v                   -- list remotes
git branch -a                   -- list branches
git branch <branchname>         -- create branch
git push -u origin <branchname> -- push new branch
git push                        -- send to default remote / branch
git add .                       -- add new files
git commit -am<message>         -- commit changes
git pull                        -- retrieve changes, fetch + merge
git rm <files>                  -- remove files
git checkout <branchname>       -- move to branch
git stash                       -- pops changes onto stack
git status                      -- get state of the project
git config --global <key> <val> -- update a setting
git config -v                   -- show settings
git mv <file> <dest>            -- rename a file
